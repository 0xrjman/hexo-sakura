读取文件信息作为视频播放起始时间

RJman · 2019-7-13 · 次阅读

需求为读取文件的创建时间作为视频的播放开始时间

读取本地文件并获取文件信息

QDate、QTime、QDateTime和QString的转换

  • 时间转QString

    QDateTime dateTime = QDateTime::currentDateTime();
QString str = dateTime.toString("yyyy/mm/dd hh:mm:ss");   //年月日 时分秒 如：2019-07-03 19:01:50

  • QString转时间

    QString  str = "2019-07-03 19:01:50";
QDateTime dateTime = QDateTime::fromString(str,"yyyy-MM-dd hh:mm:ss");

  • QDateTime、QDate、QTime 重载了<,>,可以用于比较时间

    qint64 daysTo(const QDateTime &) const; //比较天数差异
qint64 secsTo(const QDateTime &) const; //比较秒数差异
qint64 msecsTo(const QDateTime &) const; //比较毫秒数差异
int days = dEnd.daysTo(dStart);
int secs= dEnd.secsTo(dStart);
int msecs= dEnd.msecsTo(dStart);

源码

void MainWindow::openFile()
{
    QString file = QFileDialog::getOpenFileName(0, tr("Open a media file"), Config::instance().lastFile());
    if (file.isEmpty())
      return;
    info.setFile(file);
    file_size = info.size();  //文件大小,单位byte
    file_created = info.created();
    file_lastModified = info.lastModified();
    file_lastRead = info.lastRead();
    //QString str = file_lastModified.toString();
    //创建、修改和访问时间,类型QTime
    create_time = QTime::fromString(file_created.toString().mid(9,8),"hh:mm:ss");
    lastModified_time = QTime::fromString(file_lastModified.toString().mid(8,8),"hh:mm:ss");
    lastRead_time = QTime::fromString(file_lastRead.toString().mid(9,8),"hh:mm:ss");
    qDebug() << "------fileinfo------" << endl << create_time << endl
             << lastModified_time << endl << lastRead_time << endl;
    Config::instance().setLastFile(file);
    play(file);
}

修改结束的时间数值

mpEnd->setText(lastModified_time.addMSecs(mpPlayer->mediaStopPosition()).toString(QString::fromLatin1("HH:mm:ss")));

修改播放时变化的时间数值

//随着视频播放时，修改当前播放时间
void MainWindow::onPositionChange(qint64 pos)
{
    if (mpPlayer->isSeekable())
        mpTimeSlider->setValue(pos);
    QTime real_start = lastModified_time.addMSecs(pos);
    //qDebug() << "pos = " << pos << ",time = "<< lastModified_time << endl;
    mpCurrent->setText(real_start.toString(QString::fromLatin1("HH:mm:ss")));
    //setWindowTitle(QString::number(mpPlayer->statistics().video_only.currentDisplayFPS(), 'f', 2).append(" ").append(mTitle));
}

编写FileLoad类

源码

头文件 FileLoad.h

#ifndef FILELOAD_H
#define FILELOAD_H

#include <QFileInfo>
#include <QDateTime>
#include <QDate>
#include <QTime>

class FileLoad
{
public:
    FileLoad();
    void setInfo(QString &file);
    void setStartTime(int);
    QTime getStartTime();
    QFileInfo getInfo();
private:
    QFileInfo info;
    qint64 file_size;
    QDateTime file_created;
    QDateTime file_lastModified;
    QDateTime file_lastRead;
    QTime create_time;
    QTime lastModified_time;
    QTime lastRead_time;
    QTime startTime;
};
#endif // FILELOAD_H

Cpp文件：FileLoad.cpp

#include "fileload.h"

FileLoad::FileLoad()
{

}
void FileLoad::setInfo(QString &file){
    info.setFile(file);
    file_size = info.size();
    file_created = info.created();
    file_lastModified = info.lastModified();
    file_lastRead = info.lastRead();
    create_time = QTime::fromString(file_created.toString().mid(9,8),"hh:mm:ss");
    lastModified_time = QTime::fromString(file_lastModified.toString().mid(8,8),"hh:mm:ss");
    lastRead_time = QTime::fromString(file_lastRead.toString().mid(9,8),"hh:mm:ss");
    startTime = lastModified_time;
    return;
}
QFileInfo FileLoad::getInfo(){
    return info;
}
void FileLoad::setStartTime(int r)
{
    if(r == 0){
        startTime = create_time;
    }else if(r == 1){
        startTime = lastModified_time;
    }else if(r == 2){
        startTime = lastRead_time;
    }else{
        return;
    }
}
QTime FileLoad::getStartTime()
{
    return startTime;
}
